// Quiz questions
let questions = [
    {
        question: "What is the capital of France?",
        answers: ["Berlin", "Paris", "Madrid"],
        correct: 1
    },
    {
        question: "Which planet is known as the Red Planet?",
        answers: ["Mars", "Jupiter", "Venus"],
        correct: 0
    },
    {
        question: "Which language runs in a web browser?",
        answers: ["C++", "Java", "JavaScript"],
        correct: 2
    }
];

let current = 0;

// Load a question
function loadQuestion() {
    let q = questions[current];

    // Update question text using innerHTML
    document.getElementById("question").innerHTML = q.question;

    // Clear previous answers
    let answersDiv = document.getElementById("answers");
    answersDiv.innerHTML = "";

    // Clear result message
    document.getElementById("result").innerHTML = "";

    
    q.answers.forEach((answer, index) => {
        let btn = document.createElement("button");
        btn.className = "answer-btn";
        btn.innerHTML = answer;

        // Add click event
        btn.addEventListener("click", function () {
            checkAnswer(index);
        });

        
        answersDiv.appendChild(btn);
    });
}


function checkAnswer(selected) {
    let correctIndex = questions[current].correct;

    if (selected === correctIndex) {
        document.getElementById("result").innerHTML = "✅ Correct!";
    } else {
        document.getElementById("result").innerHTML = "❌ Incorrect!";
    }
}

// Next Question button event
document.getElementById("nextBtn").addEventListener("click", function () {
    current++;

    if (current < questions.length) {
        loadQuestion();
    } else {
        document.getElementById("question").innerHTML = "Quiz finished!";
        document.getElementById("answers").innerHTML = "";
        document.getElementById("result").innerHTML = "";
    }
});

// Start quiz
loadQuestion();
