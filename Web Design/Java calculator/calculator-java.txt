document.addEventListener("DOMContentLoaded", () => {
  const startBtn = document.getElementById("startBtn");
  const pricesDiv = document.getElementById("prices");
  const resultDiv = document.getElementById("result");

  startBtn.addEventListener("click", () => {
    const numItems = parseInt(document.getElementById("numItems").value);
    pricesDiv.innerHTML = ""; // clear previous inputs
    resultDiv.style.display = "none"; // hide previous result

    if (isNaN(numItems) || numItems <= 0) {
      alert("Please enter a valid number of items.");
      return;
    }

    // Create input fields for each item
    for (let i = 1; i <= numItems; i++) {
      const label = document.createElement("label");
      label.textContent = `Enter price of item ${i}: `;
      const input = document.createElement("input");
      input.type = "number";
      input.min = "0";
      input.step = "0.01";
      input.className = "priceInput";
      pricesDiv.appendChild(label);
      pricesDiv.appendChild(input);
      pricesDiv.appendChild(document.createElement("br"));
    }

    const calcButton = document.createElement("button");
    calcButton.textContent = "Calculate Total";
    calcButton.addEventListener("click", calculateTotal);
    pricesDiv.appendChild(calcButton);
  });

  function calculateTotal() {
    const priceInputs = document.querySelectorAll(".priceInput");
    let total = 0;

    for (let input of priceInputs) {
      const price = parseFloat(input.value);
      if (!isNaN(price)) {
        total += price;
      }
    }

    // Determine discount
    let discountRate = 0;
    if (total > 100) {
      discountRate = 0.10;
    } else if (total > 50) {
      discountRate = 0.05;
    }

    const discountAmount = total * discountRate;
    const finalTotal = total - discountAmount;

    // Display results
    resultDiv.style.display = "block";
    document.getElementById("total").textContent = `Total before discount: $${total.toFixed(2)}`;
    document.getElementById("discount").textContent = `Discount applied: $${discountAmount.toFixed(2)}`;
    document.getElementById("final").textContent = `Final total: $${finalTotal.toFixed(2)}`;
  }
});
